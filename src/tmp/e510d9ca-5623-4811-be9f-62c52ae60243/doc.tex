% full-width underline with text on top (used in header + notes)
\newcommand{\filledline}[1]{%
  \leavevmode
  \makebox[0pt][l]{\textcolor{linegray}{\rule{\linewidth}{0.25pt}}}%
  \makebox[\linewidth][l]{\strut #1}%
}

% medium underline (for generic inline fields)
\newcommand{\smallfilled}[1]{%
  \leavevmode
  \makebox[2.0cm][l]{\textcolor{linegray}{\rule{2.0cm}{0.25pt}}}%
  \makebox[2.0cm][l]{\strut #1}%
}

% short blank for "@= $" lines (tight)
\newcommand{\smallfilledshort}[1]{%
  \leavevmode
  \makebox[0.40cm][l]{\textcolor{linegray}{\rule{0.40cm}{0.25pt}}}%
  \makebox[0.40cm][l]{\strut #1}%
}

\newcommand{\sblank}{\underline{\hspace{1.7cm}}}

% -------- SERVICES helper macros --------
% fixed label width inside each SERVICES column
\newlength{\serviceLabelWidth}
\settowidth{\serviceLabelWidth}{\scriptsize Foaming Drain Charge \$}

% width of the blank area to the right of the label
\newlength{\serviceBlankWidth}

\newcommand{\serviceSection}[1]{%
  {\bfseries\footnotesize\textcolor{emred}{#1}\par}%
  \vspace{0.2em}%
}

% Normal label + aligned blank that can show value
\newcommand{\serviceLine}[2]{%
  \noindent\scriptsize
  \makebox[\serviceLabelWidth][l]{#1}%
  \setlength{\serviceBlankWidth}{\dimexpr\linewidth-\serviceLabelWidth\relax}%
  \makebox[0pt][l]{\textcolor{linegray}{\rule{\serviceBlankWidth}{0.25pt}}}%
  \makebox[\serviceBlankWidth][l]{\strut #2}%
  \par
}

% Red bold label + aligned blank that can show value
\newcommand{\serviceBoldLine}[2]{%
  \noindent\scriptsize
  \makebox[\serviceLabelWidth][l]{\textbf{\textcolor{emred}{#1}}}%
  \setlength{\serviceBlankWidth}{\dimexpr\linewidth-\serviceLabelWidth\relax}%
  \makebox[0pt][l]{\textcolor{linegray}{\rule{\serviceBlankWidth}{0.25pt}}}%
  \makebox[\serviceBlankWidth][l]{\strut #2}%
  \par
}

% Label ____@____=$__  (no extra spaces, each blank can show value)
\newcommand{\serviceAtCharge}[4]{%
  \noindent\scriptsize
  \mbox{#1}%
  \smallfilledshort{#2}@\smallfilledshort{#3}=\$\smallfilledshort{#4}\par
}

\newcommand{\serviceBigHeading}[1]{%
  {\bfseries\small\textcolor{emred}{#1}\par}%
}
