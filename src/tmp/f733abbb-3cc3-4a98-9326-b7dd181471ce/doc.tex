% full-width underline with text on top (used in header + notes)
\newcommand{\filledline}[1]{%
  \leavevmode
  \makebox[0pt][l]{\textcolor{linegray}{\rule{\linewidth}{0.25pt}}}%
  \makebox[\linewidth][l]{\strut #1}%
}

% medium underline (for generic inline fields)
\newcommand{\smallfilled}[1]{%
  \leavevmode
  \makebox[2.0cm][l]{\textcolor{linegray}{\rule{2.0cm}{0.25pt}}}%
  \makebox[2.0cm][l]{\strut #1}%
}

% short blank for "@  =  $" lines so they fit nicely in the column
\newcommand{\smallfilledshort}[1]{%
  \leavevmode
  \makebox[0.40cm][l]{\textcolor{linegray}{\rule{0.40cm}{0.25pt}}}%
  \makebox[0.40cm][l]{\strut #1}%
}

% small inline blank for things like "Dumpster $ _____"
\newcommand{\sblank}{\underline{\hspace{1.7cm}}}

\newcommand{\serviceSection}[1]{%
  {\bfseries\footnotesize\textcolor{emred}{#1}\par}%
  \vspace{0.2em}%
}

% Normal label + aligned blank (full width)
\newcommand{\serviceLine}[2]{%
  \noindent\scriptsize
  \makebox[\serviceLabelWidth][l]{#1}%
  \textcolor{linegray}{%
    \rule{\dimexpr\linewidth-\serviceLabelWidth\relax}{0.25pt}%
  }%
  \par
}

% Red bold label + aligned blank (full width)
\newcommand{\serviceBoldLine}[2]{%
  \noindent\scriptsize
  \makebox[\serviceLabelWidth][l]{%
    \textbf{\textcolor{emred}{#1}}%
  }%
  \textcolor{linegray}{%
    \rule{\dimexpr\linewidth-\serviceLabelWidth\relax}{0.25pt}%
  }%
  \par
}

% Label ____@____=$__  (no extra spaces)
\newcommand{\serviceAtCharge}[4]{%
  \noindent\scriptsize
  \mbox{#1}%
  \smallfilledshort{#2}@\smallfilledshort{#3}=\$\smallfilledshort{#4}\par
}
