% value on top of underline, full cell width (used in header + tables)
\newcommand{\filledline}[1]{%
  \leavevmode
  \makebox[0pt][l]{\textcolor{linegray}{\rule{\linewidth}{0.25pt}}}%
  \makebox[\linewidth][l]{\strut #1}%
}

% ======= SERVICE HELPERS (used by servicesBlockLatex) =======

% heading like "RESTROOM & HYGIENE"
\newcommand{\srvHeading}[1]{%
  {\footnotesize\bfseries\textcolor{emred}{#1}\par}%
  \vspace{0.25em}%
}

% underline of fixed width with value on top
\newcommand{\srvField}[2]{%
  % #1 width, #2 value
  \makebox[#1][l]{\textcolor{linegray}{\rule{#1}{0.25pt}}}%
  \hspace{-#1}\makebox[#1][l]{\strut #2}%
}

% type = "simple"  →  Label ______
\newcommand{\srvLineSimple}[2]{%
  \par\noindent
  {\scriptsize #1\ }%
  \srvField{2.3cm}{#2}%
}

% type = "amount"  →  Label  $ ______
\newcommand{\srvLineAmount}[3]{%
  \par\noindent
  {\scriptsize #1\ }#2\,%
  \srvField{1.7cm}{#3}%
}

% type = "triple"  →  Label  ____ @ ____ $ ____
\newcommand{\srvLineTriple}[6]{%
  \par\noindent
  {\scriptsize #1\ }%
  \srvField{0.8cm}{#2}\,{\scriptsize #3}\,%
  \srvField{0.8cm}{#4}\,{\scriptsize #5}\,%
  \srvField{0.8cm}{#6}%
}
